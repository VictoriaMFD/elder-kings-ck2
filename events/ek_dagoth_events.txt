###Dagoth begins to wake up in Red Mountain...
character_event = {
	id = dagoth.0
	hide_window = yes
	
	trigger = {
		#province_id = 
		
		# Time - depends on the game rules
		# Based on the mongol spawn event
		OR = {
			year = 3802 #2E 882 right?
			AND = {
				has_game_rule = {
					name = dagoth_ur
					value = random
				}
				total_years_played = 50
			}
		}
		NOR = { 
			year = 3805
			has_game_rule = {
				name = dagoth_ur
				value = off
			}
		}
	is_triggered_only = yes
	
	mean_time_to_happen = {
		years = 100
		
		modifier = {
			factor = 0.8
			year = 3803
		}
		modifier = {
			factor = 0.8
			year = 3804
		}
		
		# Set to historical - Has to spawn
		modifier = {
			factor = 0.01
			has_game_rule = {
				name = dagoth_ur
				value = historical
			}
		}
		
		# Set to random, delay it a bit
		modifier = {
			factor = 2
			has_game_rule = {
				name = dagoth_ur
				value = random
			}
			NOT = { total_years_played = 100 }
		}
	}
	
	immediate = {
		set_global_flag = dagoth_awakes
	}
	character_event = { id = dagoth.1 days = 3 }
	
	}
}
